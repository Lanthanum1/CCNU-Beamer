% 定义每个部分的目标文件列表
\newcommand{\sectionfiles}[1]{%
  \ifcase#1
    \input{0.tex}% 当 #1 为 0 时
  \or
    \input{1.tex}% 当 #1 为 1 时
  \or
    \input{2.tex}% 当 #1 为 2 时
  \or
    \input{3.tex}% 当 #1 为 3 时
  \else
    \typeout{Unknown section specified: #1}% 当 #1 不为 0-3 之内
  \fi
}

% 检查是否定义了 \compileSection，仅包括对应部分
\ifdefined\compileSection
  \sectionfiles{\compileSection}
\else
% 如果未定义 \compileSection，则包括所有部分
  \input{0.tex}
  \input{1.tex}
  \input{2.tex}
  \input{3.tex}
\fi
