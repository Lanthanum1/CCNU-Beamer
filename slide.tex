% 定义 sectionfiles 命令
\newcommand{\sectionfiles}[1]{%
  \edef\chapFile{#1.tex} % 构造文件名
  \IfFileExists{\chapFile}{%
    \input{\chapFile} % 如果文件存在，则包含该文件
  }{%
    \typeout{Unknown section specified: #1}% 当 #1 对应的文件不存在时输出警告
  }
}

% 检查是否定义了 \compileSection，仅包括对应部分
